83 template<>
84 GEDGraph::GraphID
85 GEDEnv<GXLNodeID, GXLLabel, GXLLabel>::
86 read_graph_from_gxl_(const std::string & dir, const std::string & filename, const std::string & graph_class, Options::GXLNodeEdgeType node_type, Options::GXLNodeEdgeType edge_type,
87  const std::unordered_set<std::string> & irrelevant_node_attributes, const std::unordered_set<std::string> & irrelevant_edge_attributes) {
88 
89  // read the file into a property tree
90  boost::property_tree::ptree root;
91  try {
92  read_xml(dir + filename, root);
93  }
94  catch (const boost::property_tree::xml_parser_error & error) {
95  throw Error(std::string("Error reading file ") + filename + ": " + error.message() + ".");
96  }
97 
98  // first sanity checks
99  if (root.count("gxl") == 0) {
100  throw Error("The file " + filename + " has the wrong format: no xml-element <gxl>.");
101  }
102  if (root.count("gxl") >= 2) {
103  throw Error("The file " + filename + " has the wrong format: more than one xml-element <gxl>.");
104  }
105  root = root.get_child("gxl");
106  if (root.count("graph") == 0) {
107  throw Error("The file " + filename + " has the wrong format: no xml-element <gxl>.<graph>");
108  }
109  if (root.count("graph") >= 2) {
110  throw Error("The file " + filename + " has the wrong format: more than one xml-element <gxl>.<graph>");
111  }
112  root = root.get_child("graph");
113 
114  // add new graph to the environment
115  GEDGraph::GraphID graph_id{add_graph(filename, graph_class)};
116 
117  // initialize local variables needed for construction of the graph
118  GXLLabel label;
119  std::string attr_name;
120  std::string attr_val;
121  GXLNodeID v_id, from, to;
122 
123  // iterate through the property tree to construct the graph
124  for (const boost::property_tree::ptree::value_type & node_or_edge : root) {
125 
126  // encountered a new vertex that has to be added to the graph
127  if (node_or_edge.first == "node") {
128  // determine the vertex ID
129  try {
130  v_id = node_or_edge.second.get<std::string>("<xmlattr>.id");
131  }
132  catch (const boost::property_tree::ptree_bad_path & error) {
133  throw Error("The file " + filename + " has the wrong format: missing xml-attribute \"id\" in element <gxl>.<graph>.<node>.");
134  }
135  // read the node label and add the vertex to the graph
136 
137  label.clear();
138  if (node_type == Options::GXLNodeEdgeType::LABELED) {
139  read_gxl_label_from_ptree_(node_or_edge, irrelevant_node_attributes, filename, label);
140  }
141  add_node(graph_id, v_id, label);
142  }
143 
144  // encountered a new edge that has to be added to the graph
145  else if (node_or_edge.first == "edge") {
146  // determine the edge's tail and head
147  try {
148  from = node_or_edge.second.get<std::string>("<xmlattr>.from");
149  }
150  catch (const boost::property_tree::ptree_bad_path & error) {
151  throw Error("The file " + filename + " has the wrong format: missing xml-attribute \"from\" in element <gxl>.<graph>.<edge>.");
152  }
153  try {
154  to = node_or_edge.second.get<std::string>("<xmlattr>.to");
155  }
156  catch (const boost::property_tree::ptree_bad_path & error) {
157  throw Error("The file " + filename + " has the wrong format: missing xml-attribute \"to\" in element <gxl>.<graph>.<edge>.");
158  }
159  // read the edge label and add the edge to the graph
160  label.clear();
161  if (edge_type == Options::GXLNodeEdgeType::LABELED) {
162  read_gxl_label_from_ptree_(node_or_edge, irrelevant_edge_attributes, filename, label);
163  }
164  add_edge(graph_id, from, to, label);
165  }
166 
167  // sanity check
168  else if (node_or_edge.first != "<xmlattr>"){
169  throw Error("The file " + filename + " has the wrong format: unexpected element <gxl>.<graph>.<" + node_or_edge.first + ">.");
170  }
171  }
172  // return ID of newly constructed graph
173  return graph_id;
174 }

178 GEDEnv<GXLNodeID, GXLLabel, GXLLabel>::
179 load_gxl_graphs(const std::string & graph_dir, const std::string & file, Options::GXLNodeEdgeType node_type, Options::GXLNodeEdgeType edge_type,
180  const std::unordered_set<std::string> & irrelevant_node_attributes, const std::unordered_set<std::string> & irrelevant_edge_attributes) {

184  read_xml(file, root);
185  }

210  }

212  throw Error("The file " + file + " has the wrong format: corrupted content in xml-attribute \"file\" of element <GraphCollection>.<graph>");
213  }
214  try {
215  graph_class = val.second.get<std::string>("<xmlattr>.class");
216  }
217  catch (const boost::property_tree::ptree_bad_path & error) {
218  throw Error("The file " + file + " has the wrong format: missing xml-attribute \"class\" in element <GraphCollection>.<graph>");
219  }
220  catch (const boost::property_tree::ptree_bad_data & error) {
221  throw Error("The file " + file + " has the wrong format: corrupted content in xml-attribute \"class\" of element <GraphCollection>.<graph>");
222  }
223  graph_ids.push_back(read_graph_from_gxl_(graph_dir, gxl_file, graph_class, node_type, edge_type, irrelevant_node_attributes, irrelevant_edge_attributes));
224  }
225  else if (val.first != "<xmlattr>") {
226  throw Error("The file " + file + " has the wrong format: unexpected element <GraphCollection>.<" + val.first + ">.");
227  }
228  }
229  return graph_ids;

232 template<class UserNodeID, class UserNodeLabel, class UserEdgeLabel>
233 void
234 GEDEnv<UserNodeID, UserNodeLabel, UserEdgeLabel>::
235 read_gxl_label_from_ptree_(const boost::property_tree::ptree::value_type & node_or_edge, const std::unordered_set<std::string> & irrelevant_attributes, const std::string & file, GXLLabel & label) {
236  std::string attr_name, attr_val;
237  std::string info(node_or_edge.first);
238  // read the attributes into the label
239  for (const boost::property_tree::ptree::value_type & attr : node_or_edge.second) {
240 
241  if (attr.first == "attr") {
242  // determine the name of the attribute
243  try {
244  attr_name = attr.second.get<std::string>("<xmlattr>.name");
245  }
246  catch (const boost::property_tree::ptree_bad_path & error) {
247  throw Error("The file " + file + " has the wrong format: missing xml-attribute \"name\" in element <gxl>.<graph>.<" + info + ">.<attr>");
248  }
249  if (irrelevant_attributes.find(attr_name) != irrelevant_attributes.end()) {
250  continue;
251  }
252  // determine the value of the attribute
253  LabelID num_attr {0};
254  for (const boost::property_tree::ptree::value_type & val : attr.second) {
255  if (val.first != "<xmlattr>") {
256  // sanity check
257  if (++num_attr > 1) {
258  throw Error("The file " + file + " has the wrong format: each element <gxl>.<graph>.<" + info + ">.<attr> has to have exactly one child. Actual number of children: " + std::to_string(num_attr));
259  }
260  try {
261  attr_val = val.second.get<std::string>("");
262  }
263  catch (const boost::property_tree::ptree_bad_path & error) {
264  throw Error("The file " + file + " has the wrong format: missing content <gxl>.<graph>.<" + info + ">.<attr>.<[TYPENAME]>.___.");
265  }
266  catch (const boost::property_tree::ptree_bad_data & error) {
267  throw Error("The file " + file + " has the wrong format: corrupted content <gxl>.<graph>.<" + info + ">.<attr>.<[TYPENAME]>.___.");
268  }
269  }
270  }
271  // add the attribute to the label
272  label[attr_name] = attr_val;
273  }
274  // sanity check
275  else if (attr.first != "<xmlattr>"){
276  throw Error("The file " + file + " has the wrong format: unexpected element <gxl>.<graph>.<" + info + ">.<" + attr.first + ">.");
277  }
278  }
279 }

290 template<class UserNodeID, class UserNodeLabel, class UserEdgeLabel>
291 void
292 GEDEnv<UserNodeID, UserNodeLabel, UserEdgeLabel>::
293 add_node(GEDGraph::GraphID graph_id, const UserNodeID & node_id, const UserNodeLabel & node_label) {
294  if (graph_id >= ged_data_.num_graphs()) {
295  throw Error("The graph " + get_graph_name(graph_id) + " has not been added to the environment.");
296  }
297  if (initialized_) {
298  throw Error("The environment is has already been initialized. Don't call add_node() after calling init().");
299  }
300  if (original_to_internal_node_ids_[graph_id].find(node_id) != original_to_internal_node_ids_[graph_id].end()) {
301  throw Error("The node " + to_string_(node_id) + " has already been added to the graph " + std::to_string(graph_id) + ": " + get_graph_name(graph_id) + ".");
302  }
303  LabelID label_id{ged_data_.node_label_to_id_(node_label)};
304  GEDGraph::NodeID internal_node_id{ged_data_.graphs_[graph_id].add_node()};
305  original_to_internal_node_ids_[graph_id][node_id] = internal_node_id;
306  internal_to_original_node_ids_[graph_id][internal_node_id] = node_id;
307  ged_data_.original_to_internal_node_ids_[graph_id][to_string_(node_id)] = internal_node_id;
308  ged_data_.internal_to_original_node_ids_[graph_id][internal_node_id] = to_string_(node_id);
309  ged_data_.graphs_[graph_id].set_label(original_to_internal_node_ids_[graph_id][node_id], label_id);
310 }

312 template<class UserNodeID, class UserNodeLabel, class UserEdgeLabel>
313 void
314 GEDEnv<UserNodeID, UserNodeLabel, UserEdgeLabel>::
315 add_edge(GEDGraph::GraphID graph_id, const UserNodeID & from, const UserNodeID & to, const UserEdgeLabel & edge_label, bool ignore_duplicates) {
316  if (graph_id >= ged_data_.num_graphs()) {
317  throw Error("The graph " + get_graph_name(graph_id) + " has not been added to the environment.");
318  }
319  if (initialized_) {
320  throw Error("The environment is has already been initialized. Don't call add_edge() after calling init().");
321  }
322  if (original_to_internal_node_ids_[graph_id].find(from) == original_to_internal_node_ids_[graph_id].end()) {
323  throw Error("The node " + to_string_(from) + " does not exist in the graph " + get_graph_name(graph_id) + ".");
324  }
325  if (original_to_internal_node_ids_[graph_id].find(to) == original_to_internal_node_ids_[graph_id].end()) {
326  throw Error("The node " + to_string_(to) + " does not exist in the graph " + get_graph_name(graph_id) + ".");
327  }
328  if (ged_data_.graphs_[graph_id].safe_is_edge(original_to_internal_node_ids_[graph_id][from], original_to_internal_node_ids_[graph_id][to])) {
329  if (ignore_duplicates) {
330  return;
331  }
332  throw Error("An edge between " + to_string_(from) + " and " + to_string_(to) + " has already been added to the graph " + get_graph_name(graph_id) + ".");
333  }
334  LabelID label_id{ged_data_.edge_label_to_id_(edge_label)};
335  GEDGraph::EdgeID edge_id{ged_data_.graphs_[graph_id].add_edge(original_to_internal_node_ids_[graph_id][from], original_to_internal_node_ids_[graph_id][to])};
336  ged_data_.graphs_[graph_id].set_label(edge_id, label_id);
337 }

423 template<class UserNodeID, class UserNodeLabel, class UserEdgeLabel>
424 void
425 GEDEnv<UserNodeID, UserNodeLabel, UserEdgeLabel>::
426 run_method(GEDGraph::GraphID g_id, GEDGraph::GraphID h_id) {
427  if (g_id >= ged_data_.num_graphs()) {
428  throw Error("The graph with ID " + std::to_string(g_id) + " has not been added to the environment.");
429  }
430  if (h_id >= ged_data_.num_graphs()) {
431  throw Error("The graph with ID " + std::to_string(h_id) + " has not been added to the environment.");
432  }
433  if (not initialized_) {
434  throw Error("The environment is uninitialized. Call init() after adding all graphs to the environment.");
435  }
436  if (not ged_method_) {
437  throw Error("No method has been set. Call set_method() before calling run().");
438  }
439  // call selected GEDApproximator and store results
440  if (ged_data_.shuffled_graph_copies_available() and (g_id == h_id)) {
441  ged_method_->run(g_id, ged_data_.id_shuffled_graph_copy(h_id));
442  }
443  else {
444  ged_method_->run(g_id, h_id);
445  }
446  std::pair<GEDGraph::GraphID, GEDGraph::GraphID> key(g_id, h_id);
447  lower_bounds_[key] = ged_method_->get_lower_bound();
448  upper_bounds_[key] = ged_method_->get_upper_bound();
449  runtimes_[key] = ged_method_->get_runtime();
450  auto it = node_maps_.find(key);
451  if (it == node_maps_.end()) {
452  node_maps_.emplace(key, ged_method_->get_node_map());
453  }
454  else {
455  it->second = ged_method_->get_node_map();
456  }
457 }

574 template<class UserNodeID, class UserNodeLabel, class UserEdgeLabel>
575 void
576 GEDEnv<UserNodeID, UserNodeLabel, UserEdgeLabel>::
577 construct_shuffled_graph_copy_(GEDGraph::GraphID graph_id) {
578  GEDGraph::GraphID copied_graph_id{add_graph(get_graph_name(graph_id), get_graph_class(graph_id))};
579  const GEDGraph & graph{ged_data_.graph(graph_id)};
580  std::vector<std::pair<UserNodeID, UserNodeLabel>> nodes;
581  for (auto node = graph.nodes().first; node != graph.nodes().second; node++) {
582  nodes.emplace_back(internal_to_original_node_ids_.at(graph_id).at(*node), ged_data_.id_to_node_label(graph.get_node_label(*node)));
583  }
584  std::random_device rng_g;
585  std::mt19937 urng_g(rng_g());
586  std::shuffle(nodes.begin(), nodes.end(), urng_g);
587  for (const auto & node : nodes) {
588  add_node(copied_graph_id, node.first, node.second);
589  }
590  for (auto edge = graph.edges().first; edge != graph.edges().second; edge++) {
591  add_edge(copied_graph_id, internal_to_original_node_ids_.at(graph_id).at(graph.tail(*edge)), internal_to_original_node_ids_.at(graph_id).at(graph.head(*edge)), ged_data_.id_to_edge_label(graph.get_edge_label(*edge)));
592  }
