67 template<class UserNodeLabel, class UserEdgeLabel>
68 void
69 LSBasedMethod<UserNodeLabel, UserEdgeLabel>::
70 ged_run_(const GEDGraph & g, const GEDGraph & h, Result & result) {
71 
72  // Initialize the method for the run between g and h.
73  ls_runtime_init_(g, h);
74 
75  // Generate the initial node maps and allocate output node maps.
76  std::vector<NodeMap> initial_node_maps;
77  generate_initial_node_maps(g, h, initial_node_maps, result);
78  for (std::size_t node_map_id = 0; node_map_id < initial_node_maps.size(); node_map_id++) {
79  result.add_node_map(g.num_nodes(), h.num_nodes());
80  }
81 
82  // Initialize lower bound.
83  if (lower_bound_method_) {
84  Result lower_bound_result;
85  lower_bound_method_->run_as_util(g, h, lower_bound_result);
86  result.set_lower_bound(std::max(result.lower_bound(), lower_bound_result.lower_bound()));
87  }
88 
89  // Parallelly run local searches starting at the initial node maps. Stop if the optimal solution has been found or if the desired number of terminated runs has been reached.
90  bool found_optimum{false};
91  std::size_t terminated_runs{0};
92 #ifdef _OPENMP
93  omp_set_num_threads(num_threads_ - 1);
94 #pragma omp parallel for if(num_threads_ > 1) schedule(dynamic)
95 #endif
96  for (std::size_t node_map_id = 0; node_map_id < initial_node_maps.size(); node_map_id++) {
97  if (not found_optimum and (terminated_runs < num_runs_from_initial_solutions_)) {
98  ls_run_from_initial_solution_(g, h, result.lower_bound(), initial_node_maps.at(node_map_id), result.node_map(node_map_id));
99 #pragma omp critical
100  {
101  found_optimum = (found_optimum or (result.lower_bound() >= result.node_map(node_map_id).induced_cost()));
102  terminated_runs++;
103  }
104  }
105  }
106 
107  // Determine the best node map.
108  result.sort_node_maps_and_set_upper_bound(num_runs_from_initial_solutions_);
109 
110 }

326 template<class UserNodeLabel, class UserEdgeLabel>
327 void
328 LSBasedMethod<UserNodeLabel, UserEdgeLabel>::
329 generate_random_initial_node_maps(const GEDGraph & g, const GEDGraph & h, std::vector<NodeMap> & initial_node_maps) {
330  std::vector<GEDGraph::NodeID> permutation_g;
331  for (auto node = g.nodes().first; node != g.nodes().second; node++) {
332  permutation_g.push_back(*node);
333  }
334  std::vector<GEDGraph::NodeID> permutation_h;
335  for (auto node = h.nodes().first; node != h.nodes().second; node++) {
336  permutation_h.push_back(*node);
337  }
338  std::size_t num_substituted_nodes{std::min(g.num_nodes(), h.num_nodes())};
339  num_substituted_nodes = std::lround(num_substituted_nodes * random_substitution_ratio_);
340  for (std::size_t counter{initial_node_maps.size()}; counter < num_initial_solutions_; counter++) {
341  std::random_device rng_g;
342  std::mt19937 urng_g(rng_g());
343  std::shuffle(permutation_g.begin(), permutation_g.end(), urng_g);
344  std::random_device rng_h;
345  std::mt19937 urng_h(rng_h());
346  std::shuffle(permutation_h.begin(), permutation_h.end(), urng_h);
347  initial_node_maps.emplace_back(g.num_nodes(), h.num_nodes());
348  for (auto node = g.nodes().first; node != g.nodes().second; node++) {
349  initial_node_maps.back().add_assignment(*node, GEDGraph::dummy_node());
350  }
351  for (auto node = h.nodes().first; node != h.nodes().second; node++) {
352  initial_node_maps.back().add_assignment(GEDGraph::dummy_node(), *node);
353  }
354  for (std::size_t pos{0}; pos < num_substituted_nodes; pos++) {
355  initial_node_maps.back().add_assignment(permutation_g[pos], permutation_h[pos]);
356  }
357  this->ged_data_.compute_induced_cost(g, h, initial_node_maps.back());
358  }
